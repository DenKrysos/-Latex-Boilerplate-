{%
	\DenKrLst@ifempty{#1}%

		\lstset{#1}%
	\fi%

		\global\let\DenKrLst@caption\undefined%
	\else%
		\gdef\DenKrLst@caption{#2}%
	\fi%
	\DenKrLst@ifempty{#3}%
		\global\let\DenKrLst@label\undefined%
	\else%
		\gdef\DenKrLst@label{#3}%
	\fi%
	\lstset{frame=none,framesep=0pt,rulesep=0pt,framerule=0pt,framexleftmargin=0pt,framexrightmargin=0pt,framextopmargin=0pt,framexbottommargin=0pt,backgroundcolor={}}%
